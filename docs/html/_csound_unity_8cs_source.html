101 public string GetPath(bool runtime = false)
102 {
103 var path = string.Empty;
104 //Debug.Log($"EnvironmentSettings GetPath from {baseFolder}");
105 switch (baseFolder)
106 {
107 case EnvironmentPathOrigin.PersistentDataPath:
108 path = GetPersistentDataPath(platform, runtime);
109 break;
110 case EnvironmentPathOrigin.StreamingAssets:
111 path = GetStreamingAssetsPath(platform, runtime);
112 break;
113 case EnvironmentPathOrigin.Plugins:
114 path = GetPluginsPath(platform, runtime);
115 break;
116 case EnvironmentPathOrigin.Absolute:
117 default:
118 break;
119 }
120 path = Path.Combine(path, suffix);
121 return path;
122 }

128 private string GetPersistentDataPath(SupportedPlatform supportedPlatform, bool runtime)
129 {
130 var res = Application.persistentDataPath;
131 switch (supportedPlatform)
132 {
133 case SupportedPlatform.MacOS:
134 res = runtime ?
135 $"~/Library/Application Support/{Application.companyName}/{Application.productName}" :
136 Application.persistentDataPath;
137 break;
138 case SupportedPlatform.Windows:
139 res = runtime ?
140 $"%userprofile%\\AppData\\LocalLow\\{Application.companyName}\\{Application.productName}" :
141 Application.persistentDataPath;
142 break;
143 case SupportedPlatform.Android:
144 res = runtime ?
145 $"/storage/emulated/0/Android/data/{Application.identifier}/files" : Application.persistentDataPath;
146 break;
147 case SupportedPlatform.iOS:

720 public static List<string> ParseCsdFileForAudioChannels(string filename)
721 {
722 if (!File.Exists(filename)) return null;
723
724 string[] fullCsdText = File.ReadAllLines(filename);
725 if (fullCsdText.Length < 1) return null;
726
727 List<string> locaAudioChannels = new List<string>();
728
729 foreach (string line in fullCsdText)
730 {
731 var trimmd = line.TrimStart();
732 if (!trimmd.Contains("chnset")) continue;
733 if (trimmd.StartsWith(";")) continue;
734 var lndx = trimmd.IndexOf("chnset");
735 var chnsetEnd = lndx + "chnset".Length + 1;
736 var prms = trimmd.Substring(chnsetEnd, trimmd.Length - chnsetEnd);
737 var split = prms.Split(',');
738 if (!split[0].StartsWith("a") && !split[0].StartsWith("ga"))
739 continue; //discard non audio variables
740 // Debug.Log("found audio channel");
741 var ach = split[1].Replace('\\', ' ').Replace('\"', ' ').Trim();
742 if (!locaAudioChannels.Contains(ach))
743 locaAudioChannels.Add(ach);
744 }
745 return locaAudioChannels;
746 }

753 public static List<CsoundChannelController> ParseCsdFile(string filename)
754 {
755 if (!File.Exists(filename)) return null;
756
757 string[] fullCsdText = File.ReadAllLines(filename);
758 if (fullCsdText.Length < 1) return null;
759
760 List<CsoundChannelController> locaChannelControllers;
761 locaChannelControllers = new List<CsoundChannelController>();
762
763 foreach (string line in fullCsdText)
764 {
765
766 if (line.Contains("</"))
767 break;
768
769 var trimmd = line.TrimStart();
770 //discard csound comments in cabbage widgets
771 if (trimmd.StartsWith(";"))
772 {
773 //Debug.Log("discarding "+line);
774 continue;
775 }
776 var control = trimmd.Substring(0, trimmd.IndexOf(" ") > -1 ? trimmd.IndexOf(" ") : 0);
777 if (control.Contains("slider") || control.Contains("button") || control.Contains("checkbox")
778 || control.Contains("groupbox") || control.Contains("form") || control.Contains("combobox"))
779 {
780 CsoundChannelController controller = new CsoundChannelController();
781 controller.type = control;
782 if (trimmd.IndexOf("caption(") > -1)
783 {
784 string infoText = trimmd.Substring(trimmd.IndexOf("caption(") + 9);
785 infoText = infoText.Substring(0, infoText.IndexOf(")") - 1);
786 controller.caption = infoText;
787 }
788
789 if (trimmd.IndexOf("text(") > -1)
790 {
791 string text = trimmd.Substring(trimmd.IndexOf("text(") + 6);
792 text = text.Substring(0, text.IndexOf(")") - 1);
793 text = text.Replace("\"", "");
794 text = text.Replace('"', new char());
795 controller.text = text;
796 if (controller.type == "combobox") //if combobox, create a range
797 {
798 char[] delimiterChars = { ',' };
799 string[] tokens = text.Split(delimiterChars);
800 controller.SetRange(1, tokens.Length, 0);
801
802 for (var o = 0; o < tokens.Length; o++)
803 {
804 tokens[o] = string.Join("", tokens[o].Split(default(string[]), System.StringSplitOptions.RemoveEmptyEntries));
805 }
806 controller.options = tokens;
807 }
808 }
809
810 if (trimmd.IndexOf("items(") > -1)
811 {
812 string text = trimmd.Substring(trimmd.IndexOf("items(") + 7);
813 text = text.Substring(0, text.IndexOf(")") - 1);
814 //TODO THIS OVERRIDES TEXT!
815 text = text.Replace("\"", "");
816 text = text.Replace('"', new char());
817 if (controller.type == "combobox")
818 {
819 char[] delimiterChars = { ',' };
820 string[] tokens = text.Split(delimiterChars);
821 controller.SetRange(1, tokens.Length, 0);
822
823 for (var o = 0; o < tokens.Length; o++)
824 {
825 tokens[o] = string.Join("", tokens[o].Split(default(string[]), System.StringSplitOptions.RemoveEmptyEntries));
826 }
827 controller.options = tokens;
828 }
829 }
830
831 if (trimmd.IndexOf("channel(") > -1)
832 {
833 string channel = trimmd.Substring(trimmd.IndexOf("channel(") + 9);
834 channel = channel.Substring(0, channel.IndexOf(")") - 1);
835 controller.channel = channel;
836 }
837
838 if (trimmd.IndexOf("range(") > -1)
839 {
840 int rangeAt = trimmd.IndexOf("range(");
841 if (rangeAt != -1)
842 {
843 string range = trimmd.Substring(rangeAt + 6);
844 range = range.Substring(0, range.IndexOf(")"));
845 char[] delimiterChars = { ',' };
846 string[] tokens = range.Split(delimiterChars);
847 for (var i = 0; i < tokens.Length; i++)
848 {
849 tokens[i] = string.Join("", tokens[i].Split(default(string[]), StringSplitOptions.RemoveEmptyEntries));
850 if (tokens[i].StartsWith("."))
851 {
852 tokens[i] = "0" + tokens[i];
853 }
854 if (tokens[i].StartsWith("-."))
855 {
856 tokens[i] = "-0" + tokens[i].Substring(2, tokens[i].Length - 2);
857 }
858 }
859 var min = float.Parse(tokens[0], CultureInfo.InvariantCulture);
860 var max = float.Parse(tokens[1], CultureInfo.InvariantCulture);
861 var val = 0f;
862 var skew = 1f;
863 var increment = 1f;
864
865 if (tokens.Length > 2)
866 {
867 val = float.Parse(tokens[2], CultureInfo.InvariantCulture);
868 }
869 if (tokens.Length > 3)
870 {
871 skew = float.Parse(tokens[3], CultureInfo.InvariantCulture);
872 }
873 if (tokens.Length > 4)
874 {
875 increment = float.Parse(tokens[4], CultureInfo.InvariantCulture);
876 }
877 // Debug.Log($"{tokens.Length}");
878 controller.SetRange(min, max, val, skew, increment);
879 }
880 }
881
882 if (line.IndexOf("value(") > -1)
883 {
884 string value = trimmd.Substring(trimmd.IndexOf("value(") + 6);
885 value = value.Substring(0, value.IndexOf(")"));
886 value = value.Replace("\"", "");
887 controller.value = value.Length > 0 ? float.Parse(value, CultureInfo.InvariantCulture) : 0;
888 if (control.Contains("combobox"))
889 {
890 //Cabbage combobox index starts from 1
891 controller.value = controller.value - 1;
892 // Debug.Log("combobox value in parse: " + controller.value);
893 }
894 }
895 locaChannelControllers.Add(controller);
896 }
897 }
898 return locaChannelControllers;
899 }

980 {
981 csound.SetChannel(channel, val);

983 // The dictionary below is used to update the serialized channels on the editor

1385 public static float Remap(float value, float from1, float to1, float from2, float to2, bool clamp = false)
1386 {

1388 if (float.IsNaN(retValue)) return from2;
1389 if (float.IsPositiveInfinity(retValue)) return to2;
1390 if (float.IsNegativeInfinity(retValue)) return from2;
1391 return clamp ? Mathf.Clamp(retValue, from2, to2) : retValue;
1392 }

1404 public static float RemapTo0to1(float value, float from, float to, float skew = 1f)
1405 {
1406 if ((to - from) == 0) return 0;

1408 var proportion = Mathf.Clamp01((value - from) / (to - from));
